Creates an array of all enabled models

 
  getEnabledModels: ->
    models = []
    for model in @models
      if model.enabled
        models.push(model)
    models
  
  @saveUndo: ->
    @undo.push(@models)
    if @undo.length > 10
      @undo.pop

  @reloadUndo: ->
    if @undo.length > 0
      @models = @undo[0]
      @undo.splice(0,1)
    
  build: ->
    length = @width*@height
    models = @getEnabledModels()
    imageData = @image.data

    if models.length == 0
      i = length
      while i--
        imageData[i] = 0
    else
      model = models.shift()
      model.build() if model.stale
      modelData = model.data
      i = length
      while i--
        imageData[i] = modelData[i]

      if models.length > 0
        for model in models
          model.build() if model.stale
          modelData = model.data
          i = length
          while i--
            imageData[i] += modelData[i]

    @residual.build(@fitsData,imageData)

    undefined
